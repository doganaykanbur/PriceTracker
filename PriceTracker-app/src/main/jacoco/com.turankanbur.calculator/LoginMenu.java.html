	/**
	 * Handles key typing events in the login username field.
	 * 
	 * @param evt the key event
	 */
	public void loginusernameKeyTyped(KeyEvent evt) {
		username = loginusername.getText() + evt.getKeyChar();
	}

	/**
	 * Handles key typing events in the login password field.
	 * 
	 * @param evt the key event
	 */
	public void loginpasswordKeyTyped(KeyEvent evt) {
		char lastChar = evt.getKeyChar();
		char[] passwordChars = loginpassword.getPassword();
		char[] updatedPasswordChars = new char[passwordChars.length + 1];
		System.arraycopy(passwordChars, 0, updatedPasswordChars, 0, passwordChars.length);
		updatedPasswordChars[passwordChars.length] = lastChar;
		password = new String(updatedPasswordChars);
	}

	/**
	 * Logs in the user with the provided username and password.
	 * 
	 * @param username the username
	 * @param password the password
	 */
	public void loginuser(String username, String password) {
		if (userDatabaseFacadeForLogin.loginUser(username, password)) {
			MainMenu mainMenu = new MainMenu(username);
			mainMenu.setVisible(true);
			this.dispose();
		} else {
			// Handle login failure
		}
	}
