	/**
	 * @brief Action performed when the jButton1 component (Add button) triggers an
	 *        action event.
	 * @param evt The action event triggered by jButton1.
	 */
	public void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {
		ingredient_name = jTextField1.getText();
		ingredient_price = jTextField2.getText();

	
		if (ingredient_name == null || ingredient_name.trim().isEmpty()) {
			JOptionPane.showMessageDialog(null, "Ingredient name cannot be empty.", "Validation Error",
					JOptionPane.ERROR_MESSAGE);
			return;
		}

		if (ingredient_price == null || ingredient_price.trim().isEmpty()) {
			JOptionPane.showMessageDialog(null, "Ingredient price cannot be empty.", "Validation Error",
					JOptionPane.ERROR_MESSAGE);
			return;
		}

		
		databaseFacade.addIngredientToDatabase(ingredient_name, ingredient_price);
	}

	/**

		this.dispose();
