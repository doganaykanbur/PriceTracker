	/**
	 * Handles the action when the exit button is clicked.
	 * 
	 * @param evt the mouse event
	 */
	public void exitbuttonMouseClicked(java.awt.event.MouseEvent evt) {

		if (frame != null) {
			frame.dispose();
		} else {
			System.exit(0);
		}
	}
